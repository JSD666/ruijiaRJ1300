def main():
  global pAct=get_actual_tcp_pose()
  global RPY=rotvec2rpy([pAct[3],pAct[4],pAct[5]])
  global r=RPY[0]
  global b=RPY[1]
  global a=RPY[2]
  global X= cos(a)*sin(b)*sin(r) - sin(a)*cos(r)
  global Y=cos(a)*cos(r) + sin(a)*sin(b)*sin(r)
  global Z= cos(b)*sin(r)
  global proportion=-0.1/Y
  speedl([X*proportion,-0.1,Z*proportion,0,0,0],0.01,100)
end
